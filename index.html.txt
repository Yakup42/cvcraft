<script>
  let docType = 'cv';
  let skills = [];
  let result = '';

  renderSkills();

  function sel(t) {
    docType = t;
    ['cv','lm','both'].forEach(x => document.getElementById('dt-'+x).classList.remove('selected'));
    document.getElementById('dt-'+t).classList.add('selected');
  }

  function renderSkills() {
    document.getElementById('skills-wrap').innerHTML = skills.map((s,i) =>
      `<div class="skill-tag">${s}<button onclick="rmSkill(${i})">×</button></div>`).join('');
  }

  function addSkill() {
    const inp = document.getElementById('skill-inp');
    const v = inp.value.trim();
    if(v && !skills.includes(v)) { skills.push(v); renderSkills(); }
    inp.value = '';
  }

  function rmSkill(i) { skills.splice(i,1); renderSkills(); }

  function setStep(n) {
    [1,2,3].forEach(i => {
      document.getElementById('s'+i).classList.add('hidden');
      const el = document.getElementById('si'+i);
      el.classList.remove('active','done');
    });
    document.getElementById('s'+n).classList.remove('hidden');
    for(let i=1;i<n;i++) document.getElementById('si'+i).classList.add('done');
    document.getElementById('si'+n).classList.add('active');
  }

  async function generate() {
    const prenom = document.getElementById('f-prenom').value || '';
    const nom = document.getElementById('f-nom').value || '';
    const poste = document.getElementById('f-poste').value || '';
    const email = document.getElementById('f-email').value || '';
    const resume = document.getElementById('f-resume').value || '';
    const exp = document.getElementById('f-exp').value || '';
    const formation = document.getElementById('f-formation').value || '';
    const comp = skills.join(', ');

    setStep(2);
    const msgs = ['Analyse de votre profil...','Rédaction du contenu...','Optimisation pour les recruteurs...','Finalisation...'];
    let mi = 0;
    const iv = setInterval(() => { if(mi<msgs.length-1){mi++;document.getElementById('load-msg').textContent=msgs[mi];} }, 1200);

    let prompt = '';
    if(docType==='cv'||docType==='both') {
      prompt = `Tu es un expert RH français. Génère un CV professionnel complet en français.\nNom: ${prenom} ${nom}\nPoste visé: ${poste}\nEmail: ${email}\nAccroche: ${resume}\nExpérience: ${exp}\nFormation: ${formation}\nCompétences: ${comp}\n\nFormat structuré avec sections: PROFIL, EXPÉRIENCES, FORMATION, COMPÉTENCES. Optimisé ATS.`;
    }
    if(docType==='lm') {
      prompt = `Tu es un expert RH français. Génère une lettre de motivation professionnelle et percutante en français.\nNom: ${prenom} ${nom}\nPoste visé: ${poste}\nProfil: ${resume}\nExpérience: ${exp}\nFormation: ${formation}\nCompétences: ${comp}\n\n3-4 paragraphes, accroche forte.`;
    }
    if(docType==='both') {
      prompt += `\n\n---\n\nGénère aussi une lettre de motivation professionnelle en 3-4 paragraphes pour ce même profil.`;
    }

    try {
      const res = await fetch('/api/generate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ prompt })
      });
      const data = await res.json();
      result = data.result || 'Erreur lors de la génération.';
    } catch(e) {
      result = 'Erreur réseau. Réessayez.';
    }

    clearInterval(iv);
    document.getElementById('res-text').textContent = result;
    const titles = {cv:'Votre CV',lm:'Votre lettre de motivation',both:'Votre CV + Lettre de motivation'};
    document.getElementById('res-title').textContent = titles[docType];
    if(docType==='both') document.getElementById('pay-section').classList.remove('hidden');
    setStep(3);
  }

  function copyText() { navigator.clipboard.writeText(result); event.target.textContent='Copié !'; setTimeout(()=>{event.target.textContent='Copier';},2000); }
  function dlTxt() {
    const a = document.createElement('a');
    a.href = URL.createObjectURL(new Blob([result],{type:'text/plain'}));
    a.download = 'cvcraft.txt'; a.click();
  }
  function pay() { alert('Stripe — sera intégré prochainement !'); }
  function goBack() { setStep(1); document.getElementById('pay-section').classList.add('hidden'); }
</script>
